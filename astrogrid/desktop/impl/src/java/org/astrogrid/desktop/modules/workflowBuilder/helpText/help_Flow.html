The flow element composes a set of activities into a parallel structure; activities in the flow will be executed simultaneously. 
    <!-- execute some activities sequentially --> 
      <flow> 
        <step name="a">
          <!-- omitted -->
        </step>
        <script>
          <!-- omitted -->
        </script>
        <step name="b">
          <!-- omitted -->
        </step>
      </flow>
